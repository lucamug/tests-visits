Queue - First-In First-Out (FIFO) paradigm

Script

function queue () {
    var queue = [];
    this.enqueue = function (value) {
      queue.push(value);
      return queue;
    };
    this.dequeue = function () {
      return queue.shift();
    }
    return this;
};
var q = queue();
document.getElementById("result").innerHTML += JSON.stringify(q.enqueue("ciao")) + "\n";
document.getElementById("result").innerHTML += JSON.stringify(q.enqueue("konnichiwa")) + "\n";
document.getElementById("result").innerHTML += JSON.stringify(q.dequeue()) + "\n";
document.getElementById("result").innerHTML += JSON.stringify(q.dequeue()) + "\n";
document.getElementById("result").innerHTML += JSON.stringify(q.dequeue()) + "\n";
